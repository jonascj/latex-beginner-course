\noindent Careless / sloppy math: 
\begin{equation}
	\int_a^b f(x)dx 
	= (\frac{1}{2}
	sin(x)\sqrt{x})^2
\end{equation}
Beautiful math:
\begin{equation}
	\int_a^b f(x) \, \mathrm{d} x
	= \left(\frac{1}{2}
	\sin(x)\sqrt{x}\right)^2
\end{equation}
